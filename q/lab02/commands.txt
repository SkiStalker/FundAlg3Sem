Load A, in.txt;          // Загружаем массив A из in.txt
Rand B, 5, 1, 100;       // Генерируем массив B из 5 случайных чисел от 1 до 100
Concat A, B;             // Конкатенируем массивы A и B
Save A, out.txt;         // Сохраняем объединённый массив A в out.txt
Print A, all;            // Выводим все элементы массива A
Sort A+;                 // Сортировка массива А
Stats A;                 // Выводим статистику для массива A
